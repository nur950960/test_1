ls = показывает содержимое папки (файлы и другие папки)
ls название_директории{py33} = показывает все файлы и папки внутри папки py33
ls -a = показывает скрытые файлы и папки
ls -l = показывает дополнительную информацию о файлах и директориях
ls -la = показывает скрытые файли и директории с подробной информацией

cd = переход в домашнюю директорию
cd ~ = переход в домашнюю директорию
cd test = переход в папку test, если такой директории нет, то выйдет ошибка
cd .. = переход на директорию выше 
cd ../.. = переход на 2 директории выше
cd - = cntr + z

mkdir test = создает папку test в текущей директории
mkdir test/test2 = создает папку test2 внутри папки test, если папка test существует
mkdir test test2 = создает папку test и test2 в текущей директории

touch file.txt = создает файл 'file' с разрешением 'txt'
touch file1.txt file2.txt = создает файл 'file1.txt' и 'file2.txt'

rm test.txt = удаляет файл test.txt
rmdir test = удаляет пустую папку test
rm -rf test = удаляет всё: файлы, директории с содержимым,
              пустые директории. Рекурсивное удаление(полное)

mv test.txt test = перемещает файл test.txt в директорию test
mv test1.txt test2.txt = переименовывает файл
                         test1.txt на test2.txt
#mv работает также с директориями

pwd(print working directory) = показывает аблсолютный путь
      (начиная с корневой папки /) до текущей директории

history = показывает все команды, которые мы вводили
          в данный терминал 
history 10 = показывает 10 последних введеных команд
             из истории команд
history | grep test = показывает команды из истории команд
                      содержавшие слово test

nano test.txt = открывает файл test.txt в 
                текстовом редакторе nano

cat test.txt = выводит содержимое файла test.txt в терминал

sudo(super user do) = права суперюзера

apt = системный установщик пакетов (ubuntu) 
brew = установщик пакетов (macos)

set = выводит список переменных окружения 
env = выводит список переменных окружения

ps = для вывода запущенных процессов

!! = используется для последней веденной команды 


export = добавляет новую переменную в переменные окружения

find = находит нужный файл, когда находишься в домашней директории 

exit = закрывает терминал

useradd = пишет команду добавления нового пользователя (без username).
